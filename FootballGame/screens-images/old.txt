    <script>

        let main_div = document.getElementById("main_div");
        let Man = document.getElementById("Man");
        let dashLine = document.getElementById("dashLine");
        const Football_img = document.getElementById('Football_img');
        const nexdClose = document.getElementById("nexdClose");
        const container = document.querySelector(".container")
        const container_inner = document.querySelector(".container-inner-one")
        const container_ground = document.getElementById("container-ground")
        let dragTxt = document.getElementById("dragTxt");
        let player1 = document.getElementById("player1");

        var screen1text = document.getElementById("screen1text")
        var screen1text2 = document.getElementById("screen1text2")
        var screen1text3 = document.getElementById("screen1text3")
        var startDiv = document.getElementById("startDiv")
        var ctaPulse = document.getElementById("ctaPulse")
        var goal = document.getElementById("goal")

        setTimeout(() => {
            screen1text.style.opacity = 0;

            screen1text.style.opacity = "all .7s";
            screen1text2.style.display = "block";
            screen1text3.style.display = "block";
            setTimeout(() => {
                screen1text3.style.animation = "ctaPulse .7s ease alternate infinite";
            }, 200);


        }, 4500);

        nexdClose.addEventListener("click", () => {
            main_div.style.display = "none";
            player1.muted = true;
            player1.pause();
        })

        screen1text3.addEventListener("click", function () {
            setTimeout(() => {
                startDiv.style.opacity = "0"
                startDiv.style.transition = "all .7s"
                loadSecondImages.style.display = "block";

                playGame();

                setTimeout(() => {
                    Man.style.animation = "floatingMan 3.5s ease-in-out infinite alternate";
                }, 2000);
            }, 200);
        })


        function playGame() {
            container_ground.style.display = "block"
            container_inner.style.display = "none";
            setTimeout(() => {
                dragTxt.style.opacity = 0;
                dragTxt.style.transition = "all .5s"
            }, 3000);

            countDown()

            callback()
        }

        function countDown() {
            var countdownElement = document.getElementById("countdown");
            for (var i = 10; i >= 1; i--) {
                (function (count) {
                    setTimeout(function () {
                        countdownElement.textContent = count;
                    }, (10 - count) * 1000);
                })(i);
            }

            // final two screens block
            setTimeout(() => {
                lastTwoScreens()
            }, 10000);
        }

        function lastTwoScreens() {
            afterCompletion.style.display = "block"
            setTimeout(() => {
                lastScreen.style.display = "block";
                player1.muted = false;
                player1.play();
                setTimeout(() => {
                    cta.style.animation = "ctaPulse .7s ease alternate infinite";
                }, 2100);
            }, 3500);
        }


        let counter = 0;
        let touchstartX = 0
        let touchendX = 0
        const newPosition = counter + (counter * 5);

        let touchCounter = 0;
        

        function callback() {
            dashLine.style.display = "block";

            Football_img.addEventListener('touchstart', e => {

                touchstartX = e.changedTouches[0].screenX;
                // touchCounter++;
                // console.log("touchCounter:", touchCounter);
                e.preventDefault();
                const touchTarget = document.elementFromPoint(e.touches[0].clientX, e.touches[0].clientY);
                if (Football_img === touchTarget) {
                    touchCounter++; // Increment touch counter
                    goal.textContent = `Football dragged: ${touchCounter} times`;
                    console.log("touchCounter :", touchCounter);
                }

                goal.textContent = `${touchCounter - 1}`;
            });

            Football_img.addEventListener('touchmove', e => {

                touchendX = e.changedTouches[0].screenX;
                // console.log(touchendX);
                if (touchendX < touchstartX) {
                    if (counter > 0 && counter <= 30) {
                        dashLine.style.transform = `rotate(${counter}deg)`;
                        dashLine.style.transition = "all .5s"

                        rightMove();
                        setTimeout(() => {
                            dashLine.style.opacity = "0"
                        }, 500);
                        setTimeout(() => {
                            dashLine.style.transform = `rotate(0deg)`;
                            dashLine.style.opacity = "1"
                        }, 2400);
                    }
                    counter++;
                }

                if (touchendX > touchstartX) {
                    if (counter < 0 && counter >= -30) {
                        dashLine.style.transform = `rotate(${counter}deg)`;
                        dashLine.style.transition = "all .5s"
                        // console.log("touchstartX", touchstartX);
                        leftMove();
                        setTimeout(() => {
                            dashLine.style.opacity = "0"
                        }, 500);
                        setTimeout(() => {
                            dashLine.style.transform = `rotate(0deg)`;
                            dashLine.style.opacity = "1"
                        }, 2400);

                    }
                    counter--;
                }

                if (touchendX == touchstartX) {
                    if (counter == 0 && counter == 0) {
                        dashLine.style.transform = `rotate(${counter}deg)`;
                        dashLine.style.transition = "all .5s"

                        centerMove();
                        setTimeout(() => {
                            dashLine.style.opacity = "0"
                        }, 500);
                        setTimeout(() => {
                            dashLine.style.transform = `rotate(0deg)`;
                            dashLine.style.opacity = "1"
                        }, 2400);
                    }
                }

            });
        }

        function rightMove() {
            const newPosition = counter + (counter * 5);

            Football_img.style.transition = "all 1s";
            Football_img.style.bottom = "45%";
            detectBottomChange();
            Football_img.style.transform = "scale(0.3)";
            Football_img.style.left = `${newPosition}px`;
            setTimeout(() => {
                Football_img.style.opacity = 0;
                Football_img.style.display = "none";
            }, 999);
            setTimeout(() => {
                // Football_img.style.transition = "all 1s";
                Football_img.style.bottom = "15%";
                Football_img.style.transform = "scale(1)";
                Football_img.style.left = 0;
                Football_img.style.right = 0;
                Football_img.style.margin = "0 auto";
                Football_img.style.opacity = 1;
                Football_img.style.display = "block";
            }, 1500);
        }

        function leftMove() {
            const newPosition = counter - (counter * 5);

            Football_img.style.transition = "all 1s";
            Football_img.style.bottom = "45%";
            detectBottomChange();
            Football_img.style.transform = "scale(0.3)";
            Football_img.style.right = `${newPosition}px`;
            setTimeout(() => {
                Football_img.style.opacity = 0;
                Football_img.style.display = "none";
            }, 999);
            setTimeout(() => {
                // Football_img.style.transition = "all 1s";
                Football_img.style.bottom = "15%";
                Football_img.style.transform = "scale(1)";
                Football_img.style.left = 0;
                Football_img.style.right = 0;
                Football_img.style.margin = "0 auto";
                Football_img.style.opacity = 1;
                Football_img.style.display = "block";
            }, 1500);
        }

        function centerMove() {
            Football_img.style.transition = "all 1s";
            Football_img.style.bottom = "60%";
            detectBottomChange();
            Football_img.style.transform = "scale(0.3)";
            setTimeout(() => {
                Football_img.style.opacity = 0;
                Football_img.style.display = "none";
            }, 999);
            setTimeout(() => {
                // Football_img.style.transition = "all 1s";
                Football_img.style.bottom = "15%";
                Football_img.style.transform = "scale(1)";
                Football_img.style.left = 0;
                Football_img.style.right = 0;
                Football_img.style.margin = "0 auto";
                Football_img.style.opacity = 1;
                Football_img.style.display = "block";
            }, 1500);

        }

        let bottomChangeCounter = 0;
        let throwAttempts = 0;
        const maxAttempts = 3;

        function detectBottomChange() {
            if (throwAttempts < maxAttempts) {
                if (Football_img.style.bottom === "45%" || Football_img.style.bottom === "60%") {
                    bottomChangeCounter++;
                    throwAttempts++; // Increment attempts on valid throw


                    // console.log("bottomChangeCounter", bottomChangeCounter);

                    switch (bottomChangeCounter) {
                        case 1:
                            displayElement("twoTxt");
                            bottomChangeCounter = 1;
                            // console.log("maxAttempts", maxAttempts);
                            break;
                        case 2:
                            displayElement("fourTxt");
                            bottomChangeCounter = 2;
                            // console.log("maxAttempts", maxAttempts);
                            break;
                        case 3:
                            displayElement("sixTxt");
                            bottomChangeCounter = 3;
                            // console.log("maxAttempts", maxAttempts);
                            break;
                        case 4:
                            displayElement("eightTxt");
                            bottomChangeCounter = 4;
                            break;
                        case 5:
                            displayElement("tenTxt");
                            bottomChangeCounter = 5;
                            break;
                        default:
                            resetElements();
                            bottomChangeCounter = 0;
                            break;
                    }
                } else {
                    bottomChangeCounter = 0;
                }
            }
        }

        // Function to display element and hide others
        function displayElement(elementId) {
            const elements = ["twoTxt", "fourTxt", "sixTxt", "eightTxt", "tenTxt", "zeroTxt"];
            elements.forEach(id => {
                const element = document.getElementById(id);
                if (element) {
                    if (id === elementId) {
                        element.style.display = "block";
                    } else {
                        element.style.display = "none";
                    }
                }
            });
        }

        // Function to reset all elements to hidden
        function resetElements() {
            const elements = ["twoTxt", "fourTxt", "sixTxt", "eightTxt", "tenTxt", "zeroTxt"];
            elements.forEach(id => {
                const element = document.getElementById(id);
                element.style.display = "none";
            });
        }



    </script>
